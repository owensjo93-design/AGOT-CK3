
#####################################################################
# EFFECTS
#####################################################################

agot_councillor_personal_steward_lifestyle_rank_up_effect = {
	random = {
		chance = var:councillor_personal_steward_lifestyle_progress
		modifier = {
			add = $SKILL$
		}
		modifier = {
			factor = councillor_personal_steward_lifestyle_progress_factor
		}
		save_temporary_scope_value_as = {
			name = ranked_up
			value = yes
		}
		save_scope_as = councillor
		random_councillor = {
			limit = {
				exists = this.cp:councillor_personal_steward
				this.cp:councillor_personal_steward = scope:councillor
			}
			save_scope_as = council_owner
		}
		save_scope_value_as = {
			name = skill_type
			value = flag:$SKILL$
		}
		if = {
			limit = { has_character_modifier = councillor_personal_steward_$SKILL$_2_modifier }
			remove_character_modifier = councillor_personal_steward_$SKILL$_2_modifier
			scope:council_owner = {
				send_interface_message = {
					type = msg_personal_steward_task_good
					title = councillor_personal_steward_background.0001.title
					desc = councillor_personal_steward_background.0001.$SKILL$_notification
					left_icon = scope:councillor
					scope:councillor = {
						switch = {
							trigger = scope:skill_type
							flag:diplomacy = {
								add_trait = diplomat
								debug_log = "A Personal Steward reached Lifestyle Rank 3, Diplomat was set"
							}
							flag:martial = {
								add_trait = strategist
								debug_log = "A Personal Steward reached Lifestyle Rank 3, Strategist was set"
							}
							flag:stewardship = {
								add_trait = architect
								debug_log = "A Personal Steward reached Lifestyle Rank 3, Architect was set"
							}
							flag:intrigue = {
								add_trait = schemer
								debug_log = "A Personal Steward reached Lifestyle Rank 3, Schemer was set"
							}
							flag:learning = {
								add_trait = scholar
								debug_log = "A Personal Steward reached Lifestyle Rank 3, Scholar was set"
							}
							fallback = { debug_log = "A Personal Steward reached Lifestyle Rank 3, but no trait was set" }
						}
					}
				}
			}
		}
		else_if = {
			limit = { has_character_modifier = councillor_personal_steward_$SKILL$_1_modifier }
			remove_character_modifier = councillor_personal_steward_$SKILL$_1_modifier
			scope:council_owner = {
				send_interface_message = {
					type = msg_personal_steward_task_good
					title = councillor_personal_steward_background.0001.title
					desc = councillor_personal_steward_background.0001.$SKILL$_notification
					left_icon = scope:councillor
					scope:councillor = {
						add_character_modifier = {
							modifier = councillor_personal_steward_$SKILL$_2_modifier
						}
					}
				}
			}
			debug_log = "A Personal Steward reached Lifestyle Rank 2"
		}
		else = {
			scope:council_owner = {
				send_interface_message = {
					type = msg_personal_steward_task_good
					title = councillor_personal_steward_background.0001.title
					desc = councillor_personal_steward_background.0001.$SKILL$_notification
					left_icon = scope:councillor
					scope:councillor = {
						add_character_modifier = {
							modifier = councillor_personal_steward_$SKILL$_1_modifier
						}
					}
				}
			}
			debug_log = "A Personal Steward reached Lifestyle Rank 1"
		}

		#Reset the progress once the character has ranked up
		set_variable = {
			name = councillor_personal_steward_lifestyle_progress
			value = 0
		}
	}
}
