agot_create_personal_coa = {
	save_scope_as = actor
	add_character_flag = has_personal_coa
	if = {
		limit = {
			scope:actor = {
				is_playable_character = yes
			}
		}
		create_character = {
			name = "Dummy"
			gender = male
			employer = scope:actor
			trait = agot_dummy_trait
			random_traits = no
			faith = scope:actor.faith
			culture = scope:actor.culture
			dynasty = generate
			save_scope_as = recipient
		}
	}
	else_if = {
		limit = {
			scope:actor = {
				is_playable_character = no
			}
		}
		title:c_ruins.holder = {
			save_scope_as = ruins
		}
		create_character = {
			name = "Dummy"
			gender = male
			employer = scope:ruins
			trait = agot_dummy_trait
			random_traits = no
			faith = scope:actor.faith
			culture = scope:actor.culture
			dynasty = generate
			save_scope_as = recipient
		}
	}
	scope:recipient = {
		house = {
			set_house_name = coa_new_name
		}
		dynasty = {
			switch = {
				trigger = scope:actor.prestige_level
				0 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 1
					}
				}
				1 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 3
					}
				}
				2 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 5
					}
				}
				3 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 7
					}
				}
				4 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 9
					}
				}
				5 = {
					add_dynasty_prestige_level = {
						subtract = dynasty_prestige_level
						add = 10
					}
				}
			}
		}
		set_variable = {
			name = my_coa_belongs_to
			value = scope:actor
		}
	}

	if = {
		limit = {
			OR = {
				scope:pcoa_plain ?= yes
				scope:pcoa_nickname ?= yes
				scope:pcoa_ancestral_quartering ?= yes
				scope:pcoa_religious ?= yes
				scope:pcoa_fourth_child ?= yes
				scope:pcoa_random ?= yes
			}
		}
		switch = {
			trigger = yes
			scope:pcoa_plain = {
				if = {
					limit = {
						scope:actor = { is_lowborn = no }
					}
					scope:recipient.house = { set_coa = scope:actor.house }
				}
				else = {
					scope:recipient.house = { generate_coa = yes }
				}
			}
			scope:pcoa_nickname = {
				agot_select_personal_coa_based_on_nickname_system = yes
			}
			scope:pcoa_ancestral_quartering = {
				agot_generate_ancestral_quartering_coa = yes
			}
			scope:pcoa_religious = {
				agot_select_special_rhllor_personal_coa = yes
			}
			scope:pcoa_fourth_child = {
				agot_select_personal_coa_based_on_birth_number = yes
			}
			scope:pcoa_random = {
				scope:recipient.house = { generate_coa = yes }
			}
		}
	}
	else = {
		if = {
			limit = {
				scope:actor = {
					has_nickname_for_personal_coa_trigger = yes #Using this because black/white/red nickname triggers are including nicknames that has no personal coa attached to.
				}
			}
			agot_select_personal_coa_based_on_nickname_system = yes
		}
		else_if = {
			limit = {
				scope:actor = {
					valid_character_for_rhllor_personal_coa_trigger = yes
				}
			}
			if = {
				limit = { has_specific_rhllor_personal_coa_trigger = yes }
				agot_select_special_rhllor_personal_coa = yes
			}
			else_if = {
				limit = { has_no_house_rhllor_personal_coa_trigger = yes }
				agot_select_no_house_rhllor_personal_coa = yes
				house = {
					remove_variable = rhllor_coa_holder
					set_variable = {
						name = rhllor_coa_holder
						value = scope:actor
					}
				}
			}
			else = {
				agot_select_rhllor_personal_coa = yes
				house = {
					remove_variable = rhllor_coa_holder
					set_variable = {
						name = rhllor_coa_holder
						value = scope:actor
					}
				}
			}
		}
		else_if = {
			limit = {
				scope:actor = {
					can_recieve_fourth_child_personal_coa_trigger = {
						ACTOR = this
					}
				}
			}
			agot_select_personal_coa_based_on_birth_number = yes
		}
		else_if = {
			limit = { scope:actor = { has_character_flag = selecting_faction_coa } }
			if = {
				limit = {
					scope:actor = {
						OR = {
							AND = {
								is_character_targaryen_63 = yes
								is_married = yes
								exists = dynasty:dynn_Velaryon.dynasty_founder.house
								scope:actor.primary_spouse.house = dynasty:dynn_Velaryon.dynasty_founder.house
							}
							is_character_targaryen_65 = yes
						}
					}
				}
				if = {
					limit = {
						is_character_targaryen_63 = yes
						is_married = yes
						exists = dynasty:dynn_Velaryon.dynasty_founder.house
						scope:actor.primary_spouse.house = dynasty:dynn_Velaryon.dynasty_founder.house
					}
					scope:recipient.house = { set_coa = rhaenrya_personal_coa }
				}
				else_if = {
					limit = {
						is_character_targaryen_65 = yes
					}
					scope:recipient.house = { set_coa = aegon_ii_personal_arms }
				}
			}
			else = {
				random_list = {
					99 = {
						trigger = {
							scope:actor = {
								has_trait = dragonrider
								any_relation = {
									type = agot_dragon
									is_alive = yes
									can_have_dragon_color_coa = yes
								}
							}
						}
						scope:recipient = { assign_dragon_color_for_coa = yes }
						scope:actor = { agot_select_faction_coa = yes }
						scope:recipient.house = { generate_coa = agot_faction_coa }
					}
					1 = {
						scope:actor = { agot_select_faction_coa = yes }
						scope:recipient.house = { generate_coa = agot_faction_coa }
					}
				}
			}
		}
		else_if = {
			limit = {
				scope:actor = {
					is_lowborn = yes
				}
			}
			scope:recipient.house = { generate_coa = yes }
		}
		else = {
			scope:recipient.house = { set_coa = scope:actor.house }
		}
	}

	scope:actor = {
		set_variable = {
			name = my_personal_coa
			value = scope:recipient
		}
		set_relation_agot_personal_coa_slave = scope:recipient
		if = {
			limit = {
				scope:actor = {
					is_ai = yes
				}
			}
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy Immediately since the AI won't be editing their own COAs
			trigger_event = agot_events_personal_coat_of_arms.0001
		}
		else = {
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy a few days after
			trigger_event = { id = agot_events_personal_coat_of_arms.0001 days = 1 }
		}
	}
}

agot_personal_coa_frame_updater_effect = {
	var:my_personal_coa.dynasty = {
		switch = {
			trigger = root.prestige_level
			0 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 1
				}
			}
			1 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 3
				}
			}
			2 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 5
				}
			}
			3 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 7
				}
			}
			4 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 9
				}
			}
			5 = {
				add_dynasty_prestige_level = {
					subtract = root.var:my_personal_coa.dynasty.dynasty_prestige_level
					add = 10
				}
			}
		}
	}
}

agot_generate_ancestral_quartering_coa = {
	scope:recipient = {
		dynasty = { set_coa = scope:actor.house }
	}
	create_character = {
		location = scope:actor.location
		faith = scope:actor.faith
		culture = scope:actor.culture
		gender = male
		dynasty = none
		save_scope_as = new_dummy
		after_creation = {
			set_house = scope:recipient.house
			house = {
				set_house_head = scope:new_dummy
			}
			scope:recipient = {
				create_dynamic_title = {
					tier = duchy
					name = placeholder
				}
				create_title_and_vassal_change = {
					type = created
					save_scope_as = change
					add_claim_on_loss = no
				}
				scope:new_title = {
					set_capital_county = title:c_ruins
					set_landless_title = yes
					set_destroy_on_gain_same_tier = yes
					set_no_automatic_claims = yes
					set_can_be_named_after_dynasty = no
					change_title_holder = {
						holder = scope:recipient
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
				title:c_ruins = {
					scope:actor = {
						if = { #Mom
							limit = {
								mother ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.mother.house }
						}
						else_if = { #Dad
							limit = {
								father ?= {
									NOT = { dynasty = dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.father.house }
						}
						else_if = { #Father's Father
							limit = {
								exists = father
								father.father ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.father.father.house }
						}
						else_if = { #Father's Mother
							limit = {
								exists = father
								father.mother ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.father.mother.house }
						}
						else_if = { #Mother's Father
							limit = {
								exists = mother
								mother.father ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.mother.father.house }
						}
						else_if = { #Mother's Mother
							limit = {
								exists = mother
								mother.mother ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.mother.mother.house }
						}
						else_if = { #Spouse
							limit = {
								primary_spouse ?= {
									NOT = { dynasty = root.dynasty }
									is_lowborn = no
								}
							}
							prev = { set_coa = root.primary_spouse.house }
						}
					}
				}
				create_cadet_branch = {}
				destroy_title = scope:new_title
				title:c_ruins = {
					set_coa = ruins_coa
				}
				scope:new_dummy = {
					death = {
						death_reason = death_vanished
					}
				}
			}
		}
	}
}

agot_create_personal_coa_for_historical_ai = {
	$ACTOR$ = {
		save_scope_as = actor
		add_character_flag = has_personal_coa
		title:c_ruins.holder = {
			save_scope_as = ruins
		}
		create_character = {
			name = "Dummy"
			gender = male
			employer = scope:ruins # AI Doesn't need to edit their COA, so send the dummy away immediately.
			trait = agot_dummy_trait
			faith = scope:actor.faith
			culture = scope:actor.culture
			dynasty = generate
			save_scope_as = recipient
		}
		scope:recipient = {
			scope:recipient.house = {
				set_house_name = coa_new_name
				set_coa = $COA_KEY$
			}
			set_variable = {
				name = my_coa_belongs_to
				value = scope:actor
			}
		}
		scope:actor = {
			set_variable = {
				name = my_personal_coa
				value = scope:recipient
			}
			set_relation_agot_personal_coa_slave = scope:recipient
			scope:recipient = { save_scope_as = exiled_dummy } # Send away the COA Dummy Immediately since the AI won't be editing their own COAs
			trigger_event = agot_events_personal_coat_of_arms.0001
		}
	}
}
agot_prune_personal_coa = {
	var:my_personal_coa ?= {
		add_character_flag = removing_personal_coa
		death = {
			death_reason = death_vanished
		}
	}
	remove_variable = my_personal_coa
	remove_character_flag = has_personal_coa
}
