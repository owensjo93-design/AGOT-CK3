agot_debug_log_effect = {
	if = {
		limit = { has_game_rule = agot_debug_on }
		debug_log = $MSG$
	}
}

agot_debug_log_event_effect = {
	if = {
		limit = { has_game_rule = agot_debug_on }
		location = {
			agot_get_terrain_effect = yes
		}
		set_variable = { name = debug_terrain value = scope:terrain }
		debug_log = $MSG$
		remove_variable = debug_terrain
	}
}
