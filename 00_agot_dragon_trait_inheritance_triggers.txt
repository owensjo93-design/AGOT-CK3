agot_dragon_inheritance_both_parents_have_active_trait = {
	$DRAGON$ = {
		has_variable_list = dragon_parent_other
		AND = {
			is_target_in_variable_list = {
				name = dragon_parent
				target = flag:$TRAIT$
			}
			is_target_in_variable_list = {
				name = dragon_parent_other
				target = flag:$TRAIT$
			}
		}
	}
}

agot_dragon_inheritance_both_parents_have_active_or_inactive_leveled_trait = {

		has_variable_list = dragon_parent_other
		OR = {
			$DRAGON$ = {
				is_target_in_variable_list = {
					name = dragon_parent
					target = flag:$TRAIT$
				}
				OR = {

						name = dragon_parent_other
						target = flag:inactive_$TRAIT$

						target = flag:$ALT_TRAIT_2$
					}
					is_target_in_variable_list = {
						name = dragon_parent_other
						target = flag:inactive_$ALT_TRAIT_2$
					}
				}
			}
			$DRAGON$ = {
				is_target_in_variable_list = {
					name = dragon_parent_other
					target = flag:$TRAIT$
				}
				OR = {
					is_target_in_variable_list = {

						name = dragon_parent
