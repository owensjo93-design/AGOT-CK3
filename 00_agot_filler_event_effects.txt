agot_filler_execute_character_effect = { # The point here is to remove penalties for certain executions
	$VICTIM$ = {
		save_scope_as = victim
	}

	$EXECUTIONER$ = {
		save_scope_as = executioner
		save_scope_as = imprisoner #Save this scope to show correct tyranny tooltip
	}

	scope:executioner = { # Placeholder for any effects on our executioner.
		if = {
			limit = {
				scope:victim = {
					has_trait = deserter
				}
			}
		}
	}

	scope:victim = {
		death = {
			death_reason = death_execution
			killer = $EXECUTIONER$
		}
	}

	execute_opinion_effect = { VICTIM = scope:victim EXECUTIONER = scope:executioner }
}

agot_find_coast_of_sea_region = {
	if = {
		limit = {
			is_commanding_army = yes
			location = {
				geographical_region = sea_sunset_sea
			}
		}
		random_county_in_region = {
			region = sunset_sea_coastal_duchies
			limit = {
				is_coastal_county = yes
			}
			save_scope_as = countyinregion
			random_province = {
				limit = {
					barony = { county = scope:countyinregion }
					is_coastal = yes
				}
				save_scope_as = shipwrecklocation
			}
		}
		root = {
			random_army = { limit = { root = { is_commanding_army = yes } } set_army_location = scope:shipwrecklocation }
			pan_camera_to_province = scope:shipwrecklocation
		}
	}
	else_if = {
		limit = {
			is_commanding_army = yes
			location = {
				is_sunset_sea_trigger = yes
			}
		}
		random_county_in_region = {
			region = sunset_sea_coastal_duchies
			limit = {
				is_coastal_county = yes
			}
			save_scope_as = countyinregion
			random_province = {
				limit = {
					barony = { county = scope:countyinregion }
					is_coastal = yes
				}
				save_scope_as = shipwrecklocation
			}
		}
		root = {
			random_army = { limit = { root = { is_commanding_army = yes } } set_army_location = scope:shipwrecklocation }
			pan_camera_to_province = scope:shipwrecklocation
		}
	}
}

agot_move_off_ship_course_effect = {
	if = {
		limit = {
			is_commanding_army = yes
			location = {
				geographical_region = sea_shivering_sea
			}
		}
		agot_random_province_in_is_shivering_sea = yes
		root = {
			random_army = { limit = { root = { is_commanding_army = yes } } set_army_location = scope:oceanprovince }
			pan_camera_to_province = scope:oceanprovince
		}
	}
	else_if = {
		limit = {
			is_commanding_army = yes
			location = {
				geographical_region = sea_sunset_sea
			}
		}
		agot_random_province_in_is_sunset_sea = yes

		root = {
			random_army = { limit = { root = { is_commanding_army = yes } } set_army_location = scope:oceanprovince }
			pan_camera_to_province = scope:oceanprovince
		}
	}
}
