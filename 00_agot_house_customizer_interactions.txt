agot_house_customizer_select_interaction = {
	category = interaction_category_diplomacy
	use_diplomatic_range = no
	icon = icon_dynasty

	is_highlighted = {
		always = yes
	}

	is_shown = {
		scope:actor = {
			is_ai = no
			has_trait = house_customizer
		}
		scope:recipient = {
			is_human = yes
			is_lowborn = no
		}
		NOT = { scope:recipient = scope:actor }
	}

	auto_accept = yes

	on_accept = {
		scope:actor = {
			trigger_event = {
				id = agot_house_customizer.0021
			}
		}
	}
}

agot_house_customizer_set_birthplace_interaction = {
	category = interaction_category_diplomacy
	use_diplomatic_range = no
	icon = child
	desc = agot_house_customizer_set_birthplace_interaction_desc

	is_highlighted = {
		always = yes
	}

	is_shown = {
		scope:actor = {
			is_ai = no
			has_character_flag = can_set_birthplace
		}
		scope:recipient = {
			is_human = yes
			is_landed = yes
			highest_held_title_tier >= tier_county
		}
	}

	auto_accept = yes

	on_accept = {
		scope:actor = {
			trigger_event = {
				id = agot_rp_event.0200
			}
		}
	}
}
