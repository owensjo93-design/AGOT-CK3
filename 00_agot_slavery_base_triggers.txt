agot_slavery_allowed_buildings_and_maa_trigger = {
	has_realm_law = slavery_allowed_law
	agot_pentos_braavos_treaty_character_trigger = no
}

agot_can_upgrade_slave_camps_trigger = {
	barony = { is_capital_barony = yes }
	OR = {
		free_building_slots > 0
		AND = {
			has_building_or_higher = agot_slave_camps_01
			NOT = { has_building = agot_slave_camps_08 }
		}
	}
}

agot_can_upgrade_any_slave_camps_trigger = {
	agot_slavery_allowed_buildings_and_maa_trigger = yes
	any_directly_owned_province = {
		agot_can_upgrade_slave_camps_trigger = yes
	}
}

		NOT = {
