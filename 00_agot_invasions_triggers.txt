has_targaryen_exile_story = {
	any_owned_story = {
		story_type = story_agot_targaryen_exile_invasion
	}
}
agot_targaryen_exiles_can_be_pinged_for_cross_content = {
	OR = {
		any_owned_story = {
			story_type = story_agot_targaryen_exile_invasion
			OR = {
				var:story_cycle_status = flag:exile_is_adult
				var:story_cycle_status = flag:exile_special_event_pre_invasion_window
				var:story_cycle_status = flag:exile_pre_invasion_window
				var:story_cycle_status = flag:exile_invasion_player_opted_to_wait
				var:story_cycle_status = flag:hatching_ceremony_failed
			}
		}
		# In cases where the Targaryens invaded, and failed, before fAegon spawns
		NOT = {
			any_owned_story = {
				story_type = story_agot_targaryen_exile_invasion
			}
		}
	}
}
invader_ally_builder_trigger = {
	OR = {
		is_allied_to = $INVADER$
		is_allied_to = $TITLE_HOLDER$
		has_truce = $TITLE_HOLDER$
		$TITLE_HOLDER$ ?= {
			has_hook = this
		}
		opinion = {
			target = $TITLE_HOLDER$
			value > 60
		}
	}
}
