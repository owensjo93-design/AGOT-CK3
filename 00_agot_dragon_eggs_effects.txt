#AMOUNT_OF_EGGS_TO_SPAWN - number between 1 and 5
#OWNER_SCOPE - artifact owner scope, here just to be sure it will always add artifact to character
#DRAGON_PARENT - parent of the egg, saved for future family setting
agot_spawn_laid_egg = {
	$OWNER_SCOPE$ = {
		set_variable = { name = spawn_eggs value = 0 }
	}
	$DRAGON_PARENT$ = {
		save_scope_as = dragon_parent
		if = {
			limit = {
				any_relation = {
					type = soulmate
					count > 0
				}
			}
			random_relation = {
				type = soulmate
				save_scope_as = dragon_parent_other
			}
		}
	}
	while = {
		limit = {
			$OWNER_SCOPE$.var:spawn_eggs < $AMOUNT_OF_EGGS_TO_SPAWN$
		}

		agot_create_weighted_dragon_egg_artifact = { OWNER = $OWNER_SCOPE$ }

		$OWNER_SCOPE$ = {
			change_variable = {
				name = spawn_eggs
				add = 1
			}
		}
	}
	$OWNER_SCOPE$ = {
		remove_variable = spawn_eggs
	}
}

create_artifact_dragon_egg_effect = {
	$OWNER$ = {
		set_artifact_rarity_illustrious = yes
		create_artifact = {
			name = dragon_egg_$COLOR$
			visuals = dragon_egg_$COLOR$
			description = dragon_egg_$COLOR$_desc
			type = $TYPE$
			wealth = scope:wealth
			quality = scope:quality
			template = general_unique_template
			history = {
				type = created_before_history
			}
			modifier = dragon_egg_modifier
			save_scope_as = newly_created_artifact
			decaying = no
		}
		scope:newly_created_artifact ?= {
			set_variable = {
				name = dragon_egg
				value = yes
			}
			save_scope_as = epic

			set_variable = { name = dragon_egg_color value = flag:$COLOR$ }

			if = {
				limit = {
					agot_dragon_population_alive = yes
				}
				set_artifact_description = dragon_egg_$COLOR$_desc_living
			}
			else = {
				set_variable = dud_egg

				if = {
					limit = {
						has_ep1_dlc_trigger = yes
					}
					reforge_artifact = {
						type = pedestal
						modifier = dud_egg_modifier
						decaying = no
					}
				}
				else = {
					reforge_artifact = {
						type = miscellaneous
						modifier = dud_egg_modifier
						decaying = no
					}
				}
			}

			if = {
				limit = {
					exists = scope:dragon_parent
					scope:dragon_parent = { has_trait = dragon }
				}
				set_variable = {
					name = dragon_parent
					value = scope:dragon_parent
				}
				save_scope_as = egg_artifact
				if = {
					limit = {
						exists = scope:dragon_parent_other
						scope:dragon_parent_other = { has_trait = dragon }
					}
					set_variable = {
						name = dragon_parent_other
						value = scope:dragon_parent_other
					}
					set_artifact_description = dragon_egg_gen_parents
					if = {
						limit = {
							NOT = { has_game_rule = agot_dragon_trait_inheritance_disabled }
						}
						agot_laid_egg_assign_parent_congenital_traits = {
							DRAGON_EGG = scope:newly_created_artifact
						}
					}
				}
				else = {
					set_artifact_description = dragon_egg_gen_parent
					if = {
						limit = {
							NOT = { has_game_rule = agot_dragon_trait_inheritance_disabled }
						}
						agot_laid_egg_assign_parent_congenital_traits = {
							DRAGON_EGG = scope:newly_created_artifact
						}
					}
				}
			}
		}
	}
}

agot_set_dragon_parents = {
	$EGG$ = {
		hidden_effect = {
			if = {
				limit = {
					has_variable = dragon_parent
				}
				var:dragon_parent = {
					if = {
						limit = {
							is_female = yes
						}
						scope:dragon = {
							set_mother = prev
							if = {
								limit = {
									exists = prev.house
								}
								set_house = prev.house
							}
						}
					}
					else = {
						scope:dragon = {
							set_father = prev
							if = {
								limit = {
									exists = prev.house
								}
								set_house = prev.house
							}
						}
					}
				}
				if = {
					limit = {
						has_variable = dragon_parent_other
					}
					var:dragon_parent_other = {
						if = {
							limit = {
								is_female = yes
							}
							scope:dragon = {
								set_mother = prev
								if = {
									limit = {
										exists = prev.house
									}
									set_house = prev.house
								}
							}
						}
						else = {
							scope:dragon = {
								set_father = prev
								if = {
									limit = {
										exists = prev.house
									}
									set_house = prev.house
								}
							}
						}
					}
				}
			}
		}
	}

# Transfer from egg to dragon before egg is destroyed
agot_transfer_parent_congenital_traits_to_dragon = {
	# Laying Parent Transfer
	if = {
		limit = {
			$DRAGON_EGG$ = {
				has_variable_list = dragon_parent
			}
		}
		$DRAGON_EGG$ = {
			every_in_list = {
				variable = dragon_parent
				$DRAGON$ = {
					add_to_variable_list = {
						name = dragon_parent
						target = prev
					}
				}
			}
		}
	}
	# Laying Parent's Mate Transfer
	if = {
		limit = {
			$DRAGON_EGG$ = {
				has_variable_list = dragon_parent_other
			}
		}
		$DRAGON_EGG$ = {
			every_in_list = {
				variable = dragon_parent_other
				$DRAGON$ = {
					add_to_variable_list = {
						name = dragon_parent_other
						target = prev
					}
				}
			}
		}
	}
}
