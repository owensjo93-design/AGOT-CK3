crown_theft_tbc_effect = {
	if = {
		limit = {
			scope:crown_theft_king = {
				is_landed = yes
				highest_held_title_tier = tier_empire
				primary_title = { NOT = { has_order_of_succession = election } }
			}
			any_succession_crisis_scenario_available = yes
		}
		scope:crown_theft_king = {
			if = {
				limit = { has_character_flag = trial_by_combat }
				remove_character_flag = trial_by_combat
			}
			trigger_event = agot_crown_theft.0001
			add_character_flag = {
				flag = ct_tbc
				days = 60
			}
		}
	}
	else_if = {
		limit = { has_character_flag = trial_by_combat }
		death = {
			killer = scope:sc_victor
			death_reason = death_duel_trial_by_combat
		}
	}
	else = {
		death = {
			killer = scope:sc_victor
			death_reason = death_duel
		}
	}
}

the_king_dies_effect = {
	if = {
		limit = { exists = scope:dying_king_tbc_event }
		if = {
			limit = { has_character_flag = trial_by_combat }
			death = {
				killer = scope:sc_victor
				death_reason = death_duel_trial_by_combat
			}
		}
		else = {
			death = {
				killer = scope:sc_victor
				death_reason = death_duel
			}
		}
	}
	else_if = {
		limit = { exists = scope:dying_king_battle_event }
		add_trait = wounded_2
		if = {
			limit = {
				exists = scope:enemy_knight
				scope:enemy_knight = { is_alive = yes }
			}
			death = {
				death_reason = death_battle
				killer = scope:enemy_knight
			}
		}
		else = {
			death = { death_reason = death_battle }
		}
	}
	else_if = {
		limit = { exists = scope:dying_king_illness_event }
		death = natural
	}
	else_if = {
		limit = { exists = scope:dying_king_siege_event }
		death = {
			death_reason = death_siege
		}
		add_trait = wounded_2
	}
	else = { death = natural }
}

								faith = faith:rhllor_fc
								faith = faith:rhllor_fots
							}
							has_rhllor_personal_coa_trigger = yes
						}
						AND = {
							OR = {
								faith = faith:rhllor_asshai
								faith = faith:rhllor_fc
								faith = faith:rhllor_fots
							}
							has_specific_rhllor_personal_coa_trigger = yes
						}
						has_character_flag = selecting_faction_coa
					}
					NOT = { has_character_flag = has_personal_coa }
				}
				save_scope_as = attacker
				agot_create_personal_coa = yes
			}
		}
		scope:crown_thief = {
			if = {
				limit = {
					OR = {
						has_nickname_for_personal_coa_trigger = yes
						any_sibling = {
							count = 3
							AND = {
								sex_same_as = scope:crown_thief
								age > scope:crown_thief.age
							}
						}
						AND = {
							OR = {
								faith = faith:rhllor_asshai
								faith = faith:rhllor_fc
								faith = faith:rhllor_fots
							}
							has_rhllor_personal_coa_trigger = yes
						}
						AND = {
							OR = {
								faith = faith:rhllor_asshai
								faith = faith:rhllor_fc
								faith = faith:rhllor_fots
							}
							has_specific_rhllor_personal_coa_trigger = yes
						}
						has_character_flag = selecting_faction_coa
					}
					NOT = { has_character_flag = has_personal_coa }
				}
				save_scope_as = defender
				agot_create_personal_coa = yes
				trigger_event = {
					id = agot_coa_events.0002
					days = 1
				}
			}
		}
	}

agot_wos_move_courtiers_to_faction_effect = {
	$COURTIER$ = { save_scope_as = this_courtier }
	$DESTINATION$ = { save_scope_as = faction_destination }
	#Add to court
	if = {
		limit = { is_playable_character = yes }
		#Characters
		if = {
			limit = {
				scope:this_courtier = {
					NOT = { is_courtier_of = scope:faction_destination }
				}
			}
			add_courtier = scope:this_courtier
		}
		#Dragons
		if = {
			limit = {
				scope:this_courtier = {
					has_trait = dragonrider
					any_relation = { type = agot_dragon is_alive = yes NOT = { is_courtier_of = scope:faction_destination } }
				}
			}
			scope:this_courtier = {
				every_relation = {
					type = agot_dragon
					limit = {
						has_trait = dragon
						is_alive = yes
						NOT = { is_courtier_of = scope:faction_destination }
					}
					save_scope_as = faction_dragons
					scope:faction_destination = {
						add_courtier = scope:faction_dragons
					}
				}
			}
		}
	}
	#Add to liege's court if this character is not landed
	else_if = {
		#Characters
		limit = { scope:faction_destination.liege = { is_playable_character = yes } }
		#Characters
		if = {
			limit = {
				scope:this_courtier = {
					NOT = { is_courtier_of = scope:faction_destination.liege }
				}
			}
			scope:faction_destination.liege = {
				add_courtier = scope:this_courtier
			}
		}
		#Dragons
		if = {
			limit = {
				scope:this_courtier = {
					has_trait = dragonrider
					any_relation = { type = agot_dragon is_alive = yes NOT = { is_courtier_of = scope:faction_destination.liege } }
				}
			}
			scope:this_courtier = {
				every_relation = {
					type = agot_dragon
					limit = {
						has_trait = dragon
						is_alive = yes
						NOT = { is_courtier_of = scope:faction_destination.liege }
					}
					save_scope_as = faction_dragons
					scope:faction_destination.liege = {
						add_courtier = scope:faction_dragons
					}
				}
			}
		}
		#Main Character's Dragons
		if = {
			limit = {
				scope:faction_destination = {
					has_trait = dragonrider
					any_relation = { type = agot_dragon is_alive = yes NOT = { is_courtier_of = scope:faction_destination.liege } }
				}
			}
			scope:faction_destination = {
				every_relation = {
					type = agot_dragon
					limit = {
						has_trait = dragon
						is_alive = yes
						NOT = { is_courtier_of = scope:faction_destination.liege }
					}
					save_scope_as = mc_dragon
					scope:faction_destination.liege = {
						add_courtier = scope:mc_dragon
					}
				}
			}
		}
	}
}

war_of_succession_number_effect = {
	if = {
		limit = {
			dynasty = { exists = var:war_of_succession_number }
		}
		dynasty = {
			change_variable = {
				name = war_of_succession_number
				add = 1
			}
		}
	}
	else_if = {
		limit = {
			dynasty = { NOT = { exists = var:war_of_succession_number } }
		}
		dynasty = {
			set_variable = {
				name = war_of_succession_number
				value = 1
			}
		}
	}
}
