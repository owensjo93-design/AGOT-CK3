agot_potential_pirate = {
	is_available_ai_adult = yes
	is_playable_character = no
	culture = { agot_is_wildling_culture = no }
	OR = {
		any_heir_title = { count = 0 }
		AND = {
			save_temporary_scope_as = potential
			any_heir_title = {
				place_in_line_of_succession = {
					target = scope:potential
					value >= 3
				}
			}
		}
	}
	OR = {
		has_trait = education_martial_prowess_3
		has_trait = education_martial_prowess_4
		has_trait = education_martial_prowess_5
	}
	prowess > 20
	martial > 15
	ai_boldness > high_positive_boldness
	agot_has_clergy_trait = no
	NOR = {
		has_trait = nightswatch
		has_trait = kingsguard
		has_trait = maester
	}
	trigger_if = {
		limit = { is_courtier = yes }
		employer = { save_temporary_scope_as = liege }
		any_traveling_family_member = {
			count = all
			courtier_allowed_to_leave_trigger = yes
		}
	}
	trigger_if = {
		limit = { is_foreign_court_or_pool_guest = yes }
		save_temporary_scope_as = guest
		host = { save_temporary_scope_as = host }
		any_traveling_family_member = {
			count = all
			guest_allowed_to_leave_trigger = yes
		}
	}
}
