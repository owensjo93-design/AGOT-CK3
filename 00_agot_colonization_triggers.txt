above_settlement_limit = {
	trigger_if = {
		limit = { highest_held_title_tier = tier_county }
		any_held_title = {
			tier = tier_county
			title_province ?= { has_holding_type = settlement_holding }
			count > 1
		}
	}
	trigger_else_if = {
		limit = { highest_held_title_tier = tier_duchy }
		any_held_title = {
			tier = tier_county
			title_province ?= { has_holding_type = settlement_holding }
			count > 2
		}
	}
	trigger_else_if = {
		limit = { highest_held_title_tier = tier_kingdom }
		any_held_title = {
			tier = tier_county
			title_province ?= { has_holding_type = settlement_holding }
			count > 3
		}
	}
	trigger_else = {
		any_held_title = {
			tier = tier_county
			title_province ?= { has_holding_type = settlement_holding }
			count > 4
		}
	}
}

not_settlement_or_wilderness = {
	title_province = {
		NOR = {
			has_holding_type = wilderness_holding
			has_holding_type = settlement_holding
		}
	}
}

region_is_colonized = {
	$TITLE$ = { save_temporary_scope_as = region_title }
	$TITLE$.holder = { save_temporary_scope_as = title_holder }
	scope:title_holder = { completely_controls = scope:region_title }
	custom_tooltip = {
		text = no_settlement_or_wilderness_counties
		$TITLE$ = {
			any_in_de_jure_hierarchy = {
				tier = tier_county
				not_settlement_or_wilderness = yes
			}
		}
	}
}
