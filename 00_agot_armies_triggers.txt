
is_enemy_army = {
	# Try to avoid using this trigger if possible - only at-war contexts should really need it.
	# It's fairly performance heavy and should be used sparingly, but it's also the only
	# way until we get a proper way to check if two armies are enemies.
	$ASKER$ = {
		OR = {
			any_war_enemy = {
				OR = {
					this = $ARMY$.army_owner
					any_vassal_or_below = {
						this = $ARMY$.army_owner
					}
				}
			}
			any_hostile_raider = {
				this = $ARMY$.army_owner
			}
		}
	}
}
