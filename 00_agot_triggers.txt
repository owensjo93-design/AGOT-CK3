agot_death_can_start_feud_trigger = {
	$VICTIM$ = { save_temporary_scope_as = victim }
	trigger_if = {
		limit = {
			exists = scope:victim
			exists = scope:victim.killer
			OR = {
				scope:victim.killer ?= { has_dead_character_flag = should_not_start_feud }
				scope:victim ?= { has_dead_character_flag = should_not_start_feud }
			}
		}
		always = no
	}
	trigger_else = {
		always = yes
	}
}
