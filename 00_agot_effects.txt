increase_variable = { # From Tobbzn
	if = { limit = { NOT = { has_variable = $NAME$ } }
		set_variable = { name = $NAME$ value = 0 }
	}
	change_variable = {
		name = $NAME$
		add = $AMOUNT$
	}
}

decrease_variable = { # From Tobbzn
	if = { limit = { NOT = { has_variable = $NAME$ } }
		set_variable = { name = $NAME$ value = 0 }
	}
	change_variable = {
		name = $NAME$
		subtract = $AMOUNT$
	}
}

#paramours
agot_become_paramour_effect = {
	scope:noble = {
		create_character_memory = {
			type = agot_got_paramour
			participants = {
				paramour = scope:paramour
			}
		}
	}
	scope:paramour = {
		add_trait = paramour
		set_relation_paramour = scope:noble
		create_character_memory = {
			type = agot_became_paramour
			participants = {
				paramour = scope:noble
			}
		}
		if = {
			limit = {
				scope:noble = { is_landed = yes }
				scope:paramour = { is_landed = no }
				NOR = {
					AND = {
						exists = scope:paramour.host
						scope:noble = scope:paramour.host
					}
					AND = {
						exists = scope:paramour.court_owner
						scope:noble = scope:paramour.court_owner
					}
				}
			}
			scope:paramour = {
				every_traveling_family_member = {
					scope:noble = { add_courtier ?= prev }
					hidden_effect = {
						return_to_court = yes
					}
				}
			}
		}
		else_if = {
			limit = {
				scope:noble = { is_landed = no }
				scope:paramour = { is_landed = no }
				NOR = {
					AND = {
						exists = scope:paramour.host
						scope:noble = scope:paramour.host
					}
					AND = {
						exists = scope:paramour.court_owner
						scope:noble = scope:paramour.court_owner
					}
				}
			}
			scope:paramour = {
				every_traveling_family_member = {
					scope:noble.court_owner = { add_courtier ?= prev }
					hidden_effect = {
						return_to_court = yes
					}
				}
			}
		}
	}
	if = {
		limit = {
			scope:noble = {
				any_secret = {
					secret_type = secret_lover
					secret_target = scope:paramour
				}
			}
		}
		scope:noble = {
			random_secret = {
				limit = {
					secret_type = secret_lover
					secret_target = scope:paramour
				}
				remove_secret = yes
			}
		}
	}
	if = {
		limit = {
			scope:paramour = {
				any_secret = {
					secret_type = secret_lover
					secret_target = scope:noble
				}
			}
		}
		scope:paramour = {
			random_secret = {
				limit = {
					secret_type = secret_lover
					secret_target = scope:noble
				}
				remove_secret = yes
			}
		}
	}
}

agot_free_captive_effect = {
	save_scope_as = captive
	if = {
		limit = {
			is_married = yes
		}
		every_spouse = {
			divorce = prev
		}
	}
	if = {
		limit = {
			exists = betrothed
		}
		if = {
			limit = {
				has_been_promised_grand_wedding = yes
			}
			break_grand_wedding_betrothal_effect = yes
		}
		else = {
			break_betrothal = betrothed
		}
	}
	if = {
		limit = {
			is_concubine = yes
		}
		this.concubinist = {
			remove_concubine = prev
		}
	}

	scope:attacker = {
		remove_list_variable = {
			name = marked_as_captive
			target = scope:captive
		}
	}
	scope:captive = {
		remove_list_variable = {
			name = marked_as_captive_by
			target = scope:attacker
		}
	}
	set_employer = scope:attacker
}

agot_pirate_convert_whole_realm_to_feudalism_effect = {
	custom_tooltip = all_pirate_holding_will_change
	if = {
		limit = {
			OR = {
				agot_is_valid_lp_feudal_government_target = yes
				any_liege_or_above = { agot_is_valid_lp_feudal_government_target = yes }
			}
			NOT = {
				any_liege_or_above = { has_trait = nightswatch }
			}
		}
		custom_tooltip = pirate_vassals_will_convert_lp
	}
	else = {
		custom_tooltip = pirate_vassals_will_convert
	}

	hidden_effect = {
		##Sort out new buildings for liege and all vassals
		every_vassal_or_below = {
			limit = { government_has_flag = government_is_pirate }
			if = {
				limit = {
					OR = {
						agot_is_valid_lp_feudal_government_target = yes
						any_liege_or_above = { agot_is_valid_lp_feudal_government_target = yes }
					}
					NOT = {
						any_liege_or_above = { has_trait = nightswatch }
					}
				}
				change_government = lp_feudal_government
			}
			else = {
				change_government = feudal_government
			}
			every_held_title = {
				limit = { has_title_law = pirate_succession_law }
				clear_title_laws = yes
			}
		}

		every_realm_province = {
			limit = {
				has_holding_type = pirate_den_holding
				NOT = { is_in_list = feudal_conversion }
			}
			add_to_list = feudal_conversion

		}
		every_vassal_or_below = {
			every_realm_province = {
				limit = {
					has_holding_type = pirate_den_holding
					NOT = { is_in_list = feudal_conversion }
				}
				add_to_list = feudal_conversion
			}
		}
		every_in_list = {
			list = feudal_conversion
			set_variable = {
				name = pirate_to_feudal_building_level
				value = combined_building_level
				days = 1
			}
			set_holding_type = castle_holding
			while = {
				limit = {
					var:pirate_to_feudal_building_level > 1
				}
				generate_building = yes
				change_variable = {
					name = pirate_to_feudal_building_level
					subtract = 1
				}
			}
			if = {
				limit = {
					has_building = generic_dragon_pit_01
				}
				remove_building = generic_dragon_pit_01
			}
			remove_variable = pirate_to_feudal_building_level
			remove_from_list = feudal_conversion
		}
	}

	if = {
		limit = {
			OR = {
				agot_is_valid_lp_feudal_government_target = yes
				any_liege_or_above = { agot_is_valid_lp_feudal_government_target = yes }
			}
			NOT = {
				any_liege_or_above = { has_trait = nightswatch }
			}
		}
		change_government = lp_feudal_government
	}
	else = {
		change_government = feudal_government
	}
	hidden_effect = {
		every_held_title = {
			limit = { has_title_law = pirate_succession_law }
			clear_title_laws = yes
		}
	}
}

agot_appoint_travel_leader_effect = {
	if = {
		limit = {
			NOT = { employs_court_position = travel_leader_court_position }
			any_courtier = {
				is_available_adult = yes
				is_valid_travel_leader_trigger = yes
			}
		}
		random_courtier = {
			limit = {
				is_available_adult = yes
				is_valid_travel_leader_trigger = yes
			}
			save_scope_as = travel_leader
		}
		appoint_court_position = {
			recipient = scope:travel_leader
			court_position = travel_leader_court_position
		}
	}
}

agot_learn_appropiate_languages = {
	if = {
		limit = {

					OR = {
						geographical_region = world_westeros_seven_kingdoms

		learn_language = language_agot_westeros
	}
}

agot_faceless_take_face = {
	$NEW_FACE$ = { save_scope_as = new_face }
	$OLD_FACE$ = { save_scope_as = old_face }
	if = {
		limit = {
			scope:new_face.age < 16
		}
		save_scope_value_as = {
			name = age_value
			value = 16
		}
	}
	else = {
		save_scope_value_as = {
			name = age_value
			value = scope:new_face.age
		}
	}

	if = {
		limit = {
			any_owned_story = {
				story_type = story_wear_face
			}
		}

		# if we have the story, we are currently wearing a face
		# we want to wear a new face, move current face to hall and swap to new

		random_owned_story = {
			limit = {
				story_type = story_wear_face
			}
			save_scope_as = story

			# In case you have reverted to your original face, we do not want to add it
			if = {
				limit = {
					NOT = {
						var:new_face = var:old_face
					}
				}

				global_var:hall_of_faces = {
					title_province = {
						add_to_variable_list = {
							name = faces
							target = scope:story.var:new_face
						}
					}
				}
			}
			set_variable = {
				name = new_face
				value = scope:new_face
			}

			change_variable = {
				name = faces_worn
				add = 1
			}
		}
	}
	else = {
		# The idea now is to make sure faces are the same when they return to the hall
		set_variable = {
			name = new_face
			value = scope:new_face
		}
		# I am also going to save the original face: If a faceless person visits the house of BAW
		# It makes sense to add their face to the hall

		# to make this work well, want to create a dummy character to store the original inheritable genetics
		create_character = {
			location = scope:location
			template = agot_faceless_character
			culture = scope:old_face.culture
			age = scope:old_face.age
			gender_female_chance = {
				value = 0
				if = {
					limit = { scope:old_face = { is_female = yes } }
					add = 100
				}
			}
			save_scope_as = dummy
			after_creation = {
				copy_inheritable_appearance_from = scope:old_face
				death = {
					death_reason = death_accident
				}
				agot_make_unpruneable = yes
			}
		}

		set_variable = {
			name = old_face
			value = scope:dummy
		}

		set_variable = {
			name = faces_worn
			value = 1
		}

		set_variable = {
			name = original_char
			value = yes
		}

		create_story = {
			type = story_wear_face
		}
	}

	hidden_effect = {
		copy_inheritable_appearance_from = scope:new_face
		set_age = scope:age_value
	}
	add_character_flag = {
		flag = taken_face_recently
		years = 5
	}

	# Remove face from the hall of faces list if in it
	global_var:hall_of_faces = {
		title_province = {
			remove_list_variable = {
				name = faces
				target = scope:new_face
			}
		}
	}

	trigger_event = agot_faceless_decision.0010
}

agot_faceless_take_face_adventure = {
	$NEW_FACE$ = { save_scope_as = new_face }
	$OLD_FACE$ = { save_scope_as = old_face }
	if = {
		limit = {
			scope:new_face.age < 16
		}
		save_scope_value_as = {
			name = age_value
			value = 16
		}
	}
	else = {
		save_scope_value_as = {
			name = age_value
			value = scope:new_face.age
		}
	}

	create_character = {
		age = scope:age_value
		#dynasty = none
		faith = scope:old_face.faith
		culture = scope:new_face.culture
		location = scope:old_face.location
		gender_female_chance = {
			value = 0
			if = {
				limit = { scope:new_face = { is_female = yes } }
				add = 100
			}
		}
		save_scope_as = new_character
		after_creation = {
			# Transfer Gold
			scope:old_face = { pay_short_term_gold = { target = scope:new_character gold = this.gold } }
			copy_inheritable_appearance_from = scope:new_face
			clear_traits = yes
			copy_traits = scope:old_face

			# The idea now is to make sure faces are the same when they return to the hall
			set_variable = {
				name = new_face
				value = scope:new_face
			}

			if = {
				limit = {
					prev = {
						any_owned_story = {
							story_type = story_wear_face
						}
					}
				}
				# old face is the original one in story

				prev = {
					random_owned_story = {
						limit = {
							story_type = story_wear_face
						}
						save_scope_as = story

						if = { # we have reverted to our original face already
							limit = {
								NOT = { var:new_face = var:old_face }
							}

							global_var:hall_of_faces = {
								title_province = {
									add_to_variable_list = {
										name = faces
										target = scope:story.var:new_face
									}
								}
							}
						}

						if = {
							limit = {
								has_variable = original_char
							}
							# We have already changed appearance, but are the original character
							# We want to preserve the original character
							scope:old_face = {
								copy_inheritable_appearance_from = scope:story.var:old_face
								set_age = scope:story.var:old_face.age
							}

							set_variable = {
								name = old_face
								value = scope:old_face
							}
						}
						var:old_face = { save_scope_as = original_character }
					}
				}

				set_variable = {
					name = old_face
					value = scope:original_character
				}

				set_variable = {
					name = faces_worn
					value = scope:story.var:faces_worn
				}

				change_variable = {
					name = faces_worn
					add = 1
				}
			}
			else = {

				# I am also going to save the original face: If a faceless person visits the house of BAW
				# It makes sense to add their face to the hall

				set_variable = {
					name = old_face
					value = scope:old_face
				}

				set_variable = {
					name = faces_worn
					value = 1
				}

			}

			# uses old and new face variables in story creation
			create_story = {
				type = story_wear_face
			}

			add_character_flag = {
				flag = taken_face_recently
				years = 5
			}
		}
	}

	# Create Adventurer Title for Rulers
	if = {
		limit = {
			has_dlc_feature = roads_to_power
			scope:old_face = { is_ruler = yes }
		}
		custom_tooltip = laamp_become_adventurer_tt
		hidden_effect = {
			scope:new_character ?= {
				create_landless_adventurer_title_effect = {
					REASON = flag:voluntary
					FLAVOR_CHAR = root
				}
			}
		}

	}

	# Transfer Artifacts
	hidden_effect = {
		scope:old_face = {
			every_character_artifact = {
				limit = { ep1_artifact_is_court_artifact_trigger = no }
				set_owner = scope:new_character
			}
		}
		# Transfer Player Character
		if = {
			limit = {
				has_multiple_players = no
				scope:old_face = { is_ai = no }
			}
			set_player_character = scope:new_character
		}
		# Kill Old Character
		scope:old_face = { death = { death_reason = death_vanished } }
	}

	# Remove face from the hall of faces list if in it
	global_var:hall_of_faces = {
		title_province = {
			remove_list_variable = {
				name = faces
				target = scope:new_face
			}
		}
	}

	scope:new_character = { trigger_event = agot_faceless_decision.0010 }
}
