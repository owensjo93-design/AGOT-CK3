agot_event_duel_common_trigger = {
	is_in_army = yes
	is_alive = yes
	location = {
		any_army_in_location = {
			is_army_in_combat = yes
		}
	}
	NOT = { is_current_dragonrider_warfare = yes }
}

maelys_ormund_duel_trigger = {
	NOT = {
		has_variable = maelys_ormund_duel
	}
	character:Blackfyre_14 = {
		is_at_same_location = character:Baratheon_75
		agot_event_duel_common_trigger = yes
	}
	character:Baratheon_75 = {
		agot_event_duel_common_trigger = yes
		OR = {
			is_at_war_with = character:Blackfyre_14
			any_liege_or_above = {
				is_at_war_with = character:Blackfyre_14
			}
		}
	}
}

	character:Osgrey_55 ?= {
		is_alive = yes
		is_married = no
		NOT = {
			has_character_flag = went_into_exile
			has_character_flag = at_war_with_rohanne
		}
	}
	character:Webber_13 ?= {
		NOT = {
			has_character_flag = admitted_guilt
		}
	}
}
