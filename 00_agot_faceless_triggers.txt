agot_can_afford_faceless_price = {
	OR = {
		agot_can_afford_faceless_gold_price = yes
		agot_can_afford_faceless_child_price = yes
		agot_can_afford_faceless_artifact_price = yes
		agot_can_afford_faceless_blood_price = yes
	}
}

agot_can_afford_faceless_gold_price = {
	trigger_if = {
		limit = {
			NOT = { has_character_flag = faceless_payed_gold_price }
		}
		custom_tooltip = {
			text = faceless_can_afford_gold_trigger_third
			gold > 300
		}
	}
	trigger_else = {
		custom_tooltip = {
			text = faceless_will_accept_gold_trigger_third
			always = no
		}
	}
}

agot_can_afford_faceless_child_price = {
	custom_tooltip = {
		text = faceless_payed_child_price.tt
		NOT = { has_character_flag = faceless_payed_child_price }
	}
	NOT = { has_character_flag = faceless_payed_child_price }
	trigger_if = {
		limit = {
			any_child = {
				count > 1
			}
			any_child = {
				count <= 3
			}
		}

		any_child = {
			age <= 10
			num_of_bad_genetic_traits = 0
			OR = {
				intrigue >= 5
				martial >= 5
			}
		}
	}
	trigger_else = {
		always = no
	}
}

agot_can_afford_faceless_artifact_price = {
	NOT = { has_character_flag = faceless_payed_artifact_price }
	any_character_artifact = {
		OR = {
			has_variable = dragon_egg
			has_variable = valyrian_steel
			artifact_type = dragon_horn
		}
	}
}

agot_can_afford_faceless_blood_price = {
	age < 50
	NOR = {
		has_trait = incapable
		has_trait = depressed
	}
	num_of_bad_genetic_traits = 0
}
